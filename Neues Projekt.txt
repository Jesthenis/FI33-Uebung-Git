import datetime

class ProjektManager:
    """Eine einfache Klasse zur Verwaltung von Aufgaben."""
    
    def __init__(self, name):
        self.name = name
        self.aufgaben = []

    def aufgabe_hinzufuegen(self, aufgabe):
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M")
        self.aufgaben.append({"titel": aufgabe, "zeit": timestamp})
        print(f"✅ Aufgabe '{aufgabe}' wurde hinzugefügt.")

    def zeige_aufgaben(self):
        print(f"\n--- Aufgabenliste für {self.name} ---")
        if not self.aufgaben:
            print("Keine Aufgaben vorhanden.")
        for i, aufg in enumerate(self.aufgaben, 1):
            print(f"{i}. {aufg['titel']} (Erstellt am: {aufg['zeit']})")

def main():
    # Instanz der Klasse erstellen
    mein_planer = ProjektManager("Mein Python-Projekt")

    try:
        # Beispielhafter Ablauf
        mein_planer.aufgabe_hinzufuegen("Git Repository initialisieren")
        mein_planer.aufgabe_hinzufuegen("Python Skript schreiben")
        
        # Aufgaben anzeigen
        mein_planer.zeige_aufgaben()
        
    except Exception as e:
        print(f"Ein Fehler ist aufgetreten: {e}")

# Der Standard-Einstiegspunkt in Python
if __name__ == "__main__":
    main()
